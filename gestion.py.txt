import streamlit as st
import pandas as pd
from datetime import datetime
import os

# Configuración de archivos
ARCHIVO_EXCEL = 'base_datos.xlsx'
HOJA_GESTIONES = 'Historico_Gestiones'

# Función para cargar los datos del Excel
def cargar_datos():
    if os.path.exists(ARCHIVO_EXCEL):
        return pd.read_excel(ARCHIVO_EXCEL, sheet_name=0)
    else:
        st.error("No se encontró el archivo Excel.")
        return None

# Función para guardar la gestión
def guardar_gestion(id_cliente, cliente, tipificacion, comentario):
    nueva_gestion = pd.DataFrame({
        'Fecha': [datetime.now().strftime("%Y-%m-%d %H:%M:%S")],
        'ID_Cliente': [id_cliente],
        'Nombre': [cliente],
        'Tipificacion': [tipificacion],
        'Comentarios': [comentario]
    })
    
    # Lógica para añadir al Excel sin borrar lo anterior
    with pd.ExcelWriter(ARCHIVO_EXCEL, engine='openpyxl', mode='a', if_sheet_exists='overlay') as writer:
        try:
            existente = pd.read_excel(ARCHIVO_EXCEL, sheet_name=HOJA_GESTIONES)
            df_final = pd.concat([existente, nueva_gestion], ignore_index=True)
            df_final.to_excel(writer, sheet_name=HOJA_GESTIONES, index=False)
        except:
            nueva_gestion.to_excel(writer, sheet_name=HOJA_GESTIONES, index=False)

# --- INTERFAZ ---
st.set_page_config(page_title="Gestión de Cartera", layout="wide")
st.title("📊 Panel de Gestión de Clientes")

df = cargar_datos()

if df is not None:
    # Buscador rápido
    busqueda = st.text_input("Buscar por nombre o ID del cliente")
    df_filtrado = df[df.astype(str).apply(lambda x: busqueda.lower() in x.str.lower().values, axis=1)]

    col1, col2 = st.columns([1, 1])

    with col1:
        st.subheader("Datos del Cliente")
        if not df_filtrado.empty:
            # Seleccionamos el primer resultado de la búsqueda
            seleccionado = st.selectbox("Seleccione el registro exacto:", df_filtrado.index, 
                                        format_func=lambda x: f"{df.iloc[x]['Cliente']} - ID: {df.iloc[x]['ID']}")
            cliente_data = df.iloc[seleccionado]
            st.json(cliente_data.to_dict()) # Muestra todos los campos del Excel
        else:
            st.info("Utiliza el buscador para encontrar un cliente.")

    with col2:
        st.subheader("Registrar Nueva Gestión")
        if not df_filtrado.empty:
            tipos = ["Compromiso de Pago", "No Contesta", "Ilocalizable", "Ya pagó", "Reclamo en trámite"]
            tipificacion = st.selectbox("Tipificación de la llamada", tipos)
            comentario = st.text_area("Notas adicionales")
            
            if st.button("Guardar Gestión"):
                guardar_gestion(cliente_data['ID'], cliente_data['Cliente'], tipificacion, comentario)
                st.success(f"Gestión registrada para {cliente_data['Cliente']}")
        else:
            st.write("Selecciona un cliente para habilitar la gestión.")

def login():
    if "autenticado" not in st.session_state:
        st.session_state.autenticado = False

    if not st.session_state.autenticado:
        st.title("🔐 Acceso al Sistema")
        user = st.text_input("Usuario")
        password = st.text_input("Contraseña", type="password")
        if st.button("Entrar"):
            if user == USUARIO_PRO and password == CLAVE_PRO:
                st.session_state.autenticado = True
                st.rerun()
            else:
                st.error("Credenciales incorrectas")
        return False
    return True
